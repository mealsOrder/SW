Title: To pointer to array
==============================================================
	
{
	int i = 123;
	int *pi;
	pi = &i; // o  (int*) == (int * const)
	
	
	
	int a[3] = {10,1,2}; // 표현식
	
	>> int[3] a; 		 // 표준식
	
	int[] a;
	int  *p;
	
	p = a; // o 양쪽다 1차원
	
	<항등식>
	a[0] == *a ; // o
	*p == p[0] ; // o
	
	p[0] = 100;
	p[1] = 101;
	p[2] = 102;
	
	-> 위처럼 하면 
	a[0] = 100;
	a[1] = 101;
	a[2] = 102;
	배열 a의 값이 바뀐다!
	
}
